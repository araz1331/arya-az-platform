def check_for_attacks(user_input):
    """
    Acts as a security tripwire. Returns True if a prompt injection is detected.
    """
    if not user_input:
        return False
        
    text = user_input.lower()
    
    # The "Blacklist" of known hacker phrases
    forbidden_phrases = [
        "ignore all previous",
        "system prompt",
        "debug mode",
        "developer mode",
        "override",
        "core rules",
        "internal instructions",
        "foundational prompt",
        "privacy firewall",
        "source code",
        "forget everything"
    ]
    
    # Check if any forbidden phrase is in the user's message
    for phrase in forbidden_phrases:
        if phrase in text:
            return True # Attack detected!
            
    return False # Safe to process
